#include <stdio.h>

int main() {
    int pp, sisauang, tabungan, uangbebas, bensin, jajan;
    float tabung = 0.2;
    printf("Rencana Keuangan Mingguan Neza\n");
    printf("\n");
    printf("Pemasukan minggu ini : ");
    scanf("%d", &pp);
    printf("Total Bensin minggu ini : ");
    scanf("%d", &bensin);
    printf("Total jajan minggu ini : ");
    scanf("%d", &jajan);
    sisauang = (pp - jajan - bensin);
    if (sisauang > 0) printf("\nSisa uang setelah pengeluaran : Rp%d", sisauang);
    else if (sisauang < 0) printf("\nHayo sisa uangmu minus : Rp%d", sisauang);
    tabungan = tabung * sisauang;
    if (tabungan > 0) printf("\nKamu bisa nabung : Rp%d", tabungan);
    else if (tabungan < 0) printf("\nTabunganmu minus : Rp%d", tabungan);
    uangbebas = (sisauang - tabungan);
    if (uangbebas > 0) printf("\nSELAMAT NEZA, UANG KAMU SISA Rp%d", uangbebas);
    else if (uangbebas < 0) printf("\nKAMU GABISA NABUNG NEZA!!!");

    return 0;
}
